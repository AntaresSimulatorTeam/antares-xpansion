169 int append_in(SolverAbstract::Ptr containingSolver_p,
170 std::string const &prefix_p = "") const {
171 // simply increment the columns indices
172 IntVector newmindex(
173 std::get<Attribute::INT_VECTOR>(_data)[IntVectorAttribute::MINDEX]);
174 int nbExistingCols(containingSolver_p->get_ncols());
175 for (auto &i : newmindex) {
176 i += nbExistingCols;
177 }
178
179 // rename variables
180 std::string prefix_l =
181 (prefix_p != "") ? prefix_p : ("prob" + std::to_string(appendCNT));
182 std::vector<std::string> newNames;
183 newNames.resize(_colNames.size());
184 std::transform(_colNames.begin(), _colNames.end(), newNames.begin(),
185 [&prefix_l](std::string varName_p) -> std::string {
186 return prefix_l + varName_p;
187 });
188
189 std::vector<int> mstart(
190 std::get<Attribute::INT_VALUE>(_data)[IntAttribute::NCOLS], 0);
191 solver_addcols(
192 *containingSolver_p,
193 std::get<Attribute::DBL_VECTOR>(_data)[DblVectorAttribute::OBJ], mstart,
194 IntVector(0, 0), DblVector(0, 0.0),
195 std::get<Attribute::DBL_VECTOR>(_data)[DblVectorAttribute::LB],
196 std::get<Attribute::DBL_VECTOR>(_data)[DblVectorAttribute::UB],
197 std::get<Attribute::CHAR_VECTOR>(_data)[CharVectorAttribute::COLTYPE],
198 newNames);
